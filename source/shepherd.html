parseShorthand = (obj, props) ->
  if not obj?
    return obj
  else if typeof obj is 'object'
    return obj
  else
    vals = obj.split(' ')

    if vals.length > props.length
      vals[0] = vals[0..vals.length - props.length].join(' ')
      vals.splice 1, (vals.length - props.length)

    out = {}
    for prop, i in props
      out[prop] = vals[i]

    out

class Shepherd extends Evented
  constructor: (@options={}) ->
    @steps = @options.steps ? []

  addStep: (name, step) ->
    if not step?
      step = name
    else
      step.id = name

    step = extend {}, @options.defaults, step

    @steps.push new Step(@, step)

  getById: (id) ->
    for step in @steps when step.id is id
      return step

  next: =>
    index = @steps.indexOf(@currentStep)

    if index is @steps.length - 1
      @hide index
      @trigger 'complete'
    else
      @show(index + 1)

  back: =>
    index = @steps.indexOf(@currentStep)

    @show(index - 1)

  cancel: =>
    @currentStep?.cancel()

    @trigger 'cancel'

  hide: =>
    @currentStep?.hide()

    @trigger 'hide'

  show: (key=0) ->
    if @currentStep
      @currentStep.hide()

    if typeof key is 'string'
      next = @getById key
    else
      next = @steps[key]

    if next
      @trigger 'shown', {step: next, previous: @currentStep}

      @currentStep = next
      next.show()

  start: ->
    @currentStep = null
    @next()

window.Shepherd = Shepherd
